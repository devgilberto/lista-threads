🔹 Bibliotecas
#include <stdio.h>


→ Permite usar funções de entrada e saída (tipo printf, fscanf, fopen)
#include <stdlib.h>


→ Permite usar funções como malloc e free (memória dinâmica)
#include <pthread.h>


→ Biblioteca para trabalhar com threads
________________


🔹 Definição de threads
#define NUM_THREADS 2


→ Define que vamos usar 2 threads (fixo)
________________


🔹 Struct Pixel
typedef struct {
    int r, g, b;
} Pixel;


→ Cria um tipo chamado Pixel com:
* r = vermelho
* g = verde
* b = azul
________________


🔹 Variáveis globais
Pixel *original;
Pixel *convertida;
int total_pixels;


→ Essas variáveis são globais, ou seja:
* todas as threads podem acessar
* original → imagem original
* convertida → imagem em cinza
* total_pixels → quantidade total de pixels
________________


🔹 Função da thread
void *converter(void *arg)


→ Função que cada thread vai executar
→ Recebe um argumento genérico (void*)
________________


int id = *(int *)arg;


→ Converte o argumento para int
→ id = número da thread (0 ou 1)
________________


🔹 Divisão do trabalho
int inicio = id * (total_pixels / NUM_THREADS);


→ Calcula onde a thread começa a trabalhar
Exemplo:
* thread 0 → começa no 0
* thread 1 → começa na metade
________________


int fim;


→ Declara onde ela termina
________________


if (id == NUM_THREADS - 1)
    fim = total_pixels;
else
    fim = (id + 1) * (total_pixels / NUM_THREADS);


→ Última thread pega até o final
→ Outras pegam só seu pedaço
________________


🔹 Loop de conversão
for (int i = inicio; i < fim; i++)


→ Loop pelos pixels que essa thread vai processar
________________


int r = original[i].r;
int g = original[i].g;
int b = original[i].b;


→ Pega os valores RGB do pixel
________________


int cinza = (int)(0.3 * r + 0.59 * g + 0.11 * b);


→ Converte para escala de cinza
(fórmula padrão)
________________


convertida[i].r = cinza;
convertida[i].g = cinza;
convertida[i].b = cinza;


→ Coloca o mesmo valor nos 3 canais → vira cinza
________________


return NULL;


→ Finaliza a thread
________________


🔹 MAIN
int main(int argc, char *argv[])


→ Função principal
→ argc = quantidade de argumentos
→ argv = argumentos
________________


if (argc != 3)


→ Verifica se o usuário passou:
* entrada
* saída
________________


printf("Uso: %s entrada.ppm saida.ppm\n", argv[0]);


→ Mostra como usar o programa
________________


🔹 Abrir arquivo
FILE *f = fopen(argv[1], "r");


→ Abre imagem de entrada
________________


if (!f)


→ Se falhar
________________


printf("Erro ao abrir arquivo\n");


→ Mostra erro
________________


🔹 Ler cabeçalho da imagem
char formato[3];
int largura, altura, maxval;


________________


fscanf(f, "%s", formato);


→ Lê tipo (P3)
________________


fscanf(f, "%d %d", &largura, &altura);


→ Lê tamanho
________________


fscanf(f, "%d", &maxval);


→ Lê valor máximo (geralmente 255)
________________


total_pixels = largura * altura;


→ Calcula total de pixels
________________


🔹 Alocação de memória
original = malloc(sizeof(Pixel) * total_pixels);
convertida = malloc(sizeof(Pixel) * total_pixels);


→ Reserva memória para as imagens
________________


🔹 Ler pixels
for (int i = 0; i < total_pixels; i++)


________________


fscanf(f, "%d %d %d",
       &original[i].r,
       &original[i].g,
       &original[i].b);


→ Lê cada pixel RGB
________________


fclose(f);


→ Fecha arquivo
________________


🔹 Criar threads
pthread_t threads[NUM_THREADS];
int ids[NUM_THREADS];


→ Vetor de threads e IDs
________________


for (int i = 0; i < NUM_THREADS; i++)


________________


ids[i] = i;


→ Define ID da thread
________________


pthread_create(&threads[i], NULL, converter, &ids[i]);


→ Cria thread e manda executar converter
________________


🔹 Esperar threads
pthread_join(threads[i], NULL);


→ Espera cada thread terminar
________________


🔹 Salvar imagem
FILE *out = fopen(argv[2], "w");


→ Abre arquivo de saída
________________


fprintf(out, "P3\n");
fprintf(out, "%d %d\n", largura, altura);
fprintf(out, "%d\n", maxval);


→ Escreve cabeçalho
________________


for (int i = 0; i < total_pixels; i++)


________________


fprintf(out, "%d %d %d\n",
        convertida[i].r,
        convertida[i].g,
        convertida[i].b);


→ Escreve pixels convertidos
________________


fclose(out);


→ Fecha arquivo
________________


🔹 Liberar memória
free(original);
free(convertida);


→ Libera memória usada
________________


return 0;


→ Programa terminou com sucesso
________________


🧠 Resumo simples
Esse programa:
1. Lê uma imagem .ppm
2. Divide os pixels entre 2 threads
3. Cada thread transforma sua parte em cinza
4. Junta tudo e salva a imagem
________________

* desenhar como as threads dividem o trabalho 👀
* ou simplificar ainda mais (nível “prova”)
